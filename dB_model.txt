TABLE info_news
       news_id   int NOT NULL AUTO_INCREMENT,
       title varchar(255) NOT NULL,
       short_description text NOT NULL,
       url Char NOT NULL,
       urlToImage Char NOT NULL,
       author varchar(120) NOT NULL,
       publishedAt   Date
       PRIMARY KEY (news_id)
